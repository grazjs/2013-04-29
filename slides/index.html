enyo.kind({ 
  name: "NewKind",
  published: { myValue: 3 },
  create: function(){
    this.inherited(arguments);
    this.myValueChanged();
  },
  // getMyValue: function() ...
  // setMyValue: function(inValue) ...
  // optional method that is called whenever setMyValue is called
  myValueChanged: function(oldValue){
   //do something
  }
});
